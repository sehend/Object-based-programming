using System;

namespace Quiz
{
    class Sound
    {
        private uint frequency; //22050 katlarý þeklinde artýyor
        private ushort channel; //Sadece 1 ve 2 deðerlerini alabildiði için enum olmasý daha mantýklý
        private ushort resolution; //Sadece 1,2,8,16,24 deðerlerini alabildiði için enum olmasý daha mantýklý

        //Ýmleç özelliðin üzerindeyken CTRL+R+E yaparak, Mutator ve Accessor metotlarýný otomatik yazabiliriz.
        #region Properties
        public uint GetFrequency()
        { return frequency; }
        public ushort GetChannel()
        { return channel; }
        public ushort GetResolution()
        { return resolution; }
        public void SetFrequency(uint f)
        {
            if (f % 22050 == 0)
            {
                frequency = f;
            }
            else
            {
                throw new Exception("Frekans, sadece 22050 ve katlarýnda deðer alabilir!");
            }
        }
        public void SetChannel(ushort c)
        {
            if(c==1 || c==2)
            {
                channel = c;
            }
            else
            {
                throw new Exception("Kanal sadece mono (1) veya stereo (2) olabilir!");
            }
        }
        public void SetResolution(ushort r)
        {
            if(r==1 || r==2 || r==8 || r==16 || r==24)
            {
                resolution = r;
            }
            else
            {
                throw new Exception("Çözünürlük sadece 1,2,8,16 veya 24 deðerlerini alabilir!");
            }
        }
        #endregion
        public Sound()
        {
            this.frequency = 22050;
            this.channel = 1;
            this.resolution = 1;
        }

        ~Sound() { }

        public bool IsStudioQuality()
        {
            //if(channel==2 && resolution==24 && frequency==88200)
            //{
            //    return true;
            //}
            //else
            //{
            //    return false;
            //}
            return (channel == 2 && resolution == 24 && frequency == 88200) ? true : false;
        }

        public uint Capacity(uint duration)
        {
            //b=dc(r/8)s;
            return (uint)(duration * channel * (resolution / 8) * frequency);
        }
    }
}
